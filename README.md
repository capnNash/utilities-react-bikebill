## Install

```bash
npm install --save iotright-utilities
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'iotright-utilities'

class Example extends Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
